How To Connect

Install the library with: go get -u github.com/hybridgroup/gobot-gpio

For Beaglebone:

package main

import (
  "github.com/hybridgroup/gobot"
  "github.com/hybridgroup/gobot-beaglebone"
  "github.com/hybridgroup/gobot-gpio"
)

func main() {
  beaglebone := new(gobotBeaglebone.Beaglebone)
  beaglebone.Name = "beaglebone"

  DirectPin := gobotGPIO.NewLed(beaglebone)
  DirectPin.Name = "DirectPin"
